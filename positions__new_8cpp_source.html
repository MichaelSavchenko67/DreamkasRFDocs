126 bool Position_new::isPrepaymentValid()
127 {
128  logDBG("%s", __PRETTY_FUNCTION__);
129 
130  bool rv = false;
131 
132  if(clc_full_prepayment)
133  {
134  rv = ((prepayment > 0) && (prepayment == total_pos_sum)) ? true : false;
135  }
136  else if(clc_prepayment)
137  {
138  rv = ((prepayment > 0) && (prepayment < total_pos_sum)) ? true : false;
139  }
140 
141  if(!rv)
142  {
143  logERR("%s prepayment is invalid!", __PRETTY_FUNCTION__);
144  }
145 
146  return rv;
147 }
