251 int Convert::crc_16_ciit_clc( const char *data, unsigned short len )
252 {
253  char x;
254  unsigned short crc = 0xFFFF;
255 
256  while (len--)
257  {
258  x = crc >> 8 ^ *data++;
259  x ^= x >> 4;
260  crc = (crc << 8) ^ ((unsigned short) (x << 12))
261  ^ ((unsigned short) (x << 5)) ^ ((unsigned short) x);
262  }
263  return crc;
