236 uint64_t Receipt_new::sum_get( int32_t mthd )
237 {

240  switch (mthd)

251  {
252  if (clc_credit_pay)

255  }

259 
260  for (auto &it : positions)
261  {
262  switch (mthd)
263  {
264  case Position_new::FULL_PREPAYMENT:
265  {

267  {
268  result += it.total_pos_sum_get();
269  }
270  break;
271  }
272  case Position_new::PREPAYMENT:
273  {
274  if (it.clc_prepayment)
275  {
276  result += it.prepayment;
277  }
278  break;
279  }
280  case Position_new::IMPREST:
281  {
282  if (it.clc_imprest)
283  {
284  result += it.total_pos_sum_get();
285  }
286  break;
287  }
288  case Position_new::FULL_CREDIT:
289  {
290  if (it.clc_full_credit)
291  {
292  result += it.total_pos_sum_get();
293  }
294  break;
295  }
296  case Position_new::CREDIT:
297  {
298  if (it.clc_credit)
299  {
300  result += it.credit;

302  break;
303  }
304  default:
305  {
306  logERR("%s:: неизвестное значение = %d", __func__, mthd);
