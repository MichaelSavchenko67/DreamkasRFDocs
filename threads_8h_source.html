292  {
293  sleep( 1 );
294  }
295 
296  l->barcodeThreadEnabled = true;
297 
298  while (l->barcodeThreadEnabled)
299  {
300  #ifdef DREAMKAS_FA
301  while(l->abExFiscal.load())

306  #endif
307 
308  l->barcodeString = l->mBarcodeScanner.readCodeString();
309  l->barcodeMutex.lock();
310  if ( !l->barcodeString.empty() )
311  {
312  l->dataMatrix.reset();
313  l->flg_barcode_rdy = true;
314  l->cvBarcode.notify_one();
315 
316  logDBG( "%s", l->barcodeString.c_str() );
317 
318  if(l->barcodeString.length() >= BarcodeScanner::DATA_MATRIX_MIN_LEN)
319  {
320  l->dataMatrix.reset(new BarcodeScanner::DATA_MATRIX);
321  if(l->mBarcodeScanner.dataMatrixParse(l->barcodeString, l->dataMatrix))
322  {
323  l->barcodeString = l->dataMatrix->gtin;
324  }
325  else
326  {
327  l->dataMatrix.reset();
328  }
329  }
330 
331  if (l->flag_scan_en)
332  {
333  l->barcode_tmp_buf = l->barcodeString;
334  l->keypad->emulateEnterKeyDown();
335  }
336  }
337  else
338  {
339  sleep(1);
340  // при пустой строке обновим данные о текущем сканере
341  l->mBarcodeScanner.identScan();
342  }
343  l->barcodeMutex.unlock();
