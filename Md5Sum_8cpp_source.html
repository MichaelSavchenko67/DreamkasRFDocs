53  // operation, processing another message block, and updating the
54  // context.
55

65  if(m_count[0] < ((UINT4)inputLen << 3)){
66  m_count[1]++;

93 void CMd5Sum::MD5Final(unsigned char* digest) // message digest
94 {
95  // MD5 finalization. Ends an MD5 message-digest operation, writing the
96  // the message digest and zeroizing the context.
97 
98  unsigned char bits[8];
99  unsigned int index, padLen;
100 
101  // Save number of bits
102  Encode (bits, m_count, 8);
103 
104  // Pad out to 56 mod 64.
105  index = (unsigned int)((m_count[0] >> 3) & 0x3f);
106  padLen = (index < 56) ? (56 - index) : (120 - index);
107  MD5Update (PADDING, padLen);
108 
109  // Append length (before padding)
110  MD5Update (bits, 8);
111 
112  // Store state in digest
113  Encode (digest, m_state, 16);
114 
115  // Zeroize sensitive information.
116  MD5_init();
117 }
