100 int JSON::getAnswerError(string jsonResStr)

102  int iResult = -1;
103  logINFO("JSON_RESULT = %s", jsonResStr.c_str());
104 
105  if(jsonResStr.compare("OK") == 0) iResult = CAB_OK;
106  else if(jsonResStr.compare("NEED_REGISTERED") == 0) iResult = CAB_NEED_REGISTERED;
107  else if(jsonResStr.compare("NEED_HANDSHAKE") == 0) iResult = CAB_NEED_HANDSHAKE;
108  else if(jsonResStr.compare("NEED_UPDATE") == 0) iResult = CAB_NEED_UPDATE;
109  else if(jsonResStr.compare("USER_FAILURE") == 0) iResult = CAB_USER_FAILURE;
110  else if(jsonResStr.compare("REQUEST_FAILURE") == 0) iResult = CAB_REQUEST_FAILURE;
111 
112  /*
113  OK Запрос выполнен успешно
114  NEED_REGISTERED Касса не зарегистрирована
115  NEED_HANDSHAKE Кассе необходимо выполнить handshake
116  NEED_UPDATE Необходимо обновить кассу
117  USER_FAILURE Ошибка польователя (не правильны пин при регистрации, etc.)
118  REQUEST_FAILURE Запрос получен сервером, но не может быть обработан (запрос не корректный)
119  */
120 
121  logINFO("Result: %d", iResult);
122  return iResult;
123 }
