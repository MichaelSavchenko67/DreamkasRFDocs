33 bool BarcodeScanner::identScan()
34 {
35  if ( type != SCANNER_OFF )
36  {
37  logINFO( "Barcode scanner is already opened!\n" );
38  return true;
39  }
40 
41  // Проверка на сканер в режиме serial usb
42  string scannerPort = NodeConflictResolver::instance()->getScanerNode();
43  if ( !scannerPort.empty() )
44  {
45  type = SCANNER_USB_SERIAL;
46  return true;
47  }
48 
49  // Проверка на сканер в режиме usb hid
50  struct stat event;

62 {
63  string code;
64  atomic<bool> shiftEnable;
65  shiftEnable.store(false);
66 
67  struct input_event ev;
68  int fd = open(usbHidPath.c_str(), O_RDONLY);
69  if ( fd > 0 )
70  {
71  const int TIMEOUT = 5000; // таймаут проверки сканера
72  const int TIMEOUT_KEYS = 100; // таймаут между нажатиями клавиш
73 
74  pollfd pfd;
75  pfd.fd = fd;
76  pfd.events = POLLIN | POLLPRI;
77 
78  while ( true )
79  {
80  // Ожидание поступления данных со сканера, или выход по таймауту
81  int ret = poll(&pfd, 1, TIMEOUT);
82  if ( ret > 0 )
83  {
84  // обрабатываем только если есть данные для чтения
85  if ( (pfd.revents & POLLIN) || (pfd.revents & POLLPRI) )
86  {
87  do
88  {
89  if ( read(fd, &ev, sizeof(ev)) > 0 )
90  {
91  if ( (ev.type == EV_KEY) && (ev.value == KEY_PRESS) )
92  {
93  switch (ev.code)
94  {
95  case KEY_0: shiftEnable.load() ? code.push_back(')') : code.push_back('0'); break;
96  case KEY_1: shiftEnable.load() ? code.push_back('!') : code.push_back('1'); break;

99  case KEY_4: shiftEnable.load() ? code.push_back('$') : code.push_back('4'); break;
100  case KEY_5: shiftEnable.load() ? code.push_back('%') : code.push_back('5'); break;
101  case KEY_6: shiftEnable.load() ? code.push_back('^') : code.push_back('6'); break;
102  case KEY_7: shiftEnable.load() ? code.push_back('&') : code.push_back('7'); break;
103  case KEY_8: shiftEnable.load() ? code.push_back('*') : code.push_back('8'); break;
104  case KEY_9: shiftEnable.load() ? code.push_back('(') : code.push_back('9'); break;
105 
106  case KEY_A: shiftEnable.load() ? code.push_back('A') : code.push_back('a'); break;
107  case KEY_B: shiftEnable.load() ? code.push_back('B') : code.push_back('b'); break;

133  case KEY_GRAVE: shiftEnable.load() ? code.push_back('~') : code.push_back('`'); break;
134  case KEY_LEFTBRACE: shiftEnable.load() ? code.push_back('{') : code.push_back('['); break;
135  case KEY_RIGHTBRACE: shiftEnable.load() ? code.push_back('}') : code.push_back(']'); break;
136  case KEY_SEMICOLON: shiftEnable.load() ? code.push_back(':') : code.push_back(';'); break;

144 
145  case KEY_LEFTSHIFT: shiftEnable.store(true); break;
146  case KEY_RIGHTSHIFT: shiftEnable.store(true); break;
147 
148  default:
149  {
150  logERR("%s UNKNOWN input code : %d ", __PRETTY_FUNCTION__, (int)ev.code);
151  break;
152  }
153  }
154  }
155 
156  if ( (ev.type == EV_KEY) && (ev.value == KEY_RELEASE) )
157  {
158  if ( (ev.code == KEY_LEFTSHIFT) || (ev.code == KEY_RIGHTSHIFT) )
159  {
160  shiftEnable.store(false);
161  }
162  }
163  }
164  else
165  {
166  logDBG(" BARCODE READER : SCANNER REMOVED !!! ");
167  code.clear();
168  break;
169  }
170 
171  } while ( poll(&pfd, 1, TIMEOUT_KEYS) > 0 );
172 
173  break;
174  }
175  else
176  {
177  logDBG(" BARCODE READER : REVENTS !POLLIN && !POLLPRI ");
178  // все что не POLLIN считаем отключеним сканера
179  code.clear();
180  type = SCANNER_OFF;
181  break;
182  }
183  }
184  else if ( ret == -1 )
185  {
186  logDBG(" BARCODE READER : ERROR POLL TIMEOUT");
187 
188  code.clear();
189  type = SCANNER_OFF;
190  break;
191  }
192  }
193 
194  close( fd );
195  }
196  else
197  {
198  code.clear();
199  type = SCANNER_OFF;
200  }
201 
202  return code;
203 }

205 string BarcodeScanner::readCodeString()
206 {
207  string code;
208 
209  switch ( type )
210  {
211  case SCANNER_OFF:
212  {
213  code.clear();
214  break;
215  }
216  case SCANNER_USB_SERIAL:
217  {
218  string scannerPort = NodeConflictResolver::instance()->getScanerNode();
219  if ( scannerPort.empty() )
220  {
221  code.clear();
222  type = SCANNER_OFF;
223  return code;
224  }
225 
226  barcodeReader.open(scannerPort);
227  if(!barcodeReader)
228  {
229  return code;
230  }
231 
232  barcodeReader >> code;
233  barcodeReader.close();
234  break;
235  }
236  case SCANNER_USB_HID:
237  {
238  code = readCodeHID();
239  break;
240  }
241  default:
242  {
243  code.clear();
244  break;
245  }
246  }
247 
248  return code;
249 }

251 bool BarcodeScanner::dataMatrixParse(const string &code, unique_ptr<DATA_MATRIX> &dataMatrix)
252 {
253  logERR("%s %-20s:\t%s", __PRETTY_FUNCTION__, "Code", code.c_str());
254 
255  if(!dataMatrix)
256  {
257  logERR("%s Nullptr!", __PRETTY_FUNCTION__);
258  return false;
259  }
260 
261  dataMatrix->clear();
262 
263  dataMatrix->rawDataMatrix = code;
264 
265  boost::regex dataMatrixTemplate("^\\(\\d{2}\\)(0{6}(?<ean8_gs1>\\d{8})|0(?<ean13_gs1>\\d{13}))\\(\\d{2}\\)(?<serial_block>.{7})|^(0{6}(?<ean8_pack>\\d{8})|0(?<ean13_pack>\\d{13}))(?<serial_pack>.{7})(?<crypto_pack>.{8})$");
266  boost::smatch match;
267 
268  bool valid = false;
269  vector<pair<string, string>> fields;
270 
271  try
272  {
273  if(boost::regex_match(code, match, dataMatrixTemplate))
274  {
275  dataMatrix->ean8Pack = match.str("ean8_pack");
276  fields.push_back({"ean8_pack", dataMatrix->ean8Pack});
277  dataMatrix->ean8Gs1 = match.str("ean8_gs1");
278  fields.push_back({"ean8_gs1", dataMatrix->ean8Gs1});
279  dataMatrix->ean13Pack = match.str("ean13_pack");
280  fields.push_back({"ean13_pack", dataMatrix->ean13Pack});
281  dataMatrix->ean13Gs1 = match.str("ean13_gs1");
282  fields.push_back({"ean13_gs1", dataMatrix->ean13Gs1});
283  dataMatrix->serialPack = match.str("serial_pack");
284  fields.push_back({"serial_pack", dataMatrix->serialPack});
285  dataMatrix->serialBlock = match.str("serial_block");
286  fields.push_back({"serial_block", dataMatrix->serialBlock});
287  dataMatrix->cryptoPack = match.str("crypto_pack");
288  fields.push_back({"crypto_pack", dataMatrix->cryptoPack});
289 
290  bool serial = (!dataMatrix->serialPack.empty() || !dataMatrix->serialBlock.empty());
291  bool ean8 = (!dataMatrix->ean8Pack.empty() || !dataMatrix->ean8Gs1.empty());
292  bool ean13 = (!dataMatrix->ean13Pack.empty() || !dataMatrix->ean13Gs1.empty());
293 
294  valid = serial;
295  if(!dataMatrix->ean8Pack.empty() || !dataMatrix->ean13Pack.empty())
296  {
297  valid = valid && !dataMatrix->cryptoPack.empty();
298  }
299  valid = valid && (ean8 || ean13);
300  }
301  }
302  catch (boost::exception &e)
303  {
304  logERR("%s\t%s", __PRETTY_FUNCTION__, boost::diagnostic_information(e).c_str());
305  return false;
306  }
307 
308  if(!valid)
309  {
310  logERR("%s\t\t\tData matrix read failed!", __PRETTY_FUNCTION__);
311  }
312  else
313  {
314  logERR("%s\t\t\tData matrix read success!", __PRETTY_FUNCTION__);
315 
316  dataMatrix->serial = dataMatrix->serialPack + dataMatrix->serialBlock;
317  fields.push_back({"Serial", dataMatrix->serial});
318  dataMatrix->gtin = dataMatrix->ean8Pack + dataMatrix->ean8Gs1
319  + dataMatrix->ean13Pack + dataMatrix->ean13Gs1;
320  fields.push_back({"GTIN", dataMatrix->gtin});
321 
322  for(const auto &field : fields)
323  {
324  if(!field.second.empty())
325  {
326  logERR("%s %-20s:\t%s", __PRETTY_FUNCTION__, field.first.c_str(), field.second.c_str());
327  }
328  }
329  }
330 
331  return valid;
332 }

334 bool BarcodeScanner::ean13Gs1Clc(const string &ean, uint8_t &gs1_13)
335 {
336  logDBG("%s", __PRETTY_FUNCTION__);
337 
338  const uint8_t EAN13_LEN = 13;
339  gs1_13 = 0;
340 
341  if(ean.length() < (EAN13_LEN - 1))
342  {
343  logERR("%s\tInvalid size!", __PRETTY_FUNCTION__);
344  return false;
345  }
346 
347  uint8_t evenSum = 0;
348  uint8_t oddSum = 0;
349  uint8_t ind = 1;
350 
351  try
352  {
353  while(ind <= (EAN13_LEN - 1))
354  {
355  int number = ean.at(ind - 1) - '0';
356  if((ind++ % 2) == 0)
357  {
358  // Чётная позиция
359  evenSum += number;
360  }
361  else
362  {
363  // Нечётная сумма
364  oddSum += number;
365  }
366  }
367  }
368  catch (exception e)
369  {
370  logERR("%s\t%s", __PRETTY_FUNCTION__, e.what());
371  return false;
372  }
373 
374  evenSum *= 3;
375  gs1_13 = (10 - ((evenSum + oddSum) % 10)) % 10;
376 
377  logDBG("%s\tGS1_13\t:\t%d", __PRETTY_FUNCTION__, gs1_13);
378 
379  return true;
380 }
