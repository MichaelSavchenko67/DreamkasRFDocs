22 struct DYNAMIC_MASK
23 {
24  unsigned short size;
25  bool *mask;
26  //--------------------------------------
27  DYNAMIC_MASK (int newSize = 1 , int index = 0 )
28  {
29  mask = new bool [newSize];
30  size = newSize;
31  index == 1111 ? memset( mask, true, newSize ) : memset( mask, false, newSize );
32  ((index < 0) || (index >= newSize)) ? 1==1 : mask[index] = true;
33  }
34  //--------------------------------------
35  DYNAMIC_MASK (const DYNAMIC_MASK &maskRight )
36  {
37  mask = new bool[maskRight.size];
38  size = maskRight.size;
39 
40  for (int i = 0; i < size; i++)
41  {
42  mask[i] = maskRight.mask[i];
43  }
44  }
45  //--------------------------------------
46  ~DYNAMIC_MASK ()
47  {
48  delete [] mask;
49  }
50  //--------------------------------------
51  DYNAMIC_MASK& operator = (const DYNAMIC_MASK &maskRight);
52  bool operator && (const DYNAMIC_MASK &maskRight);

234  inline DYNAMIC_MASK& DYNAMIC_MASK::operator=(const DYNAMIC_MASK &maskRight)
235  {
236  if ( this != &maskRight )
237  {
238  // Если размер правого больше размера левого, то увеличиваем левое до размера правого
239  if(size < maskRight.size)
240  {
241  delete[] mask;
242  mask = new bool[maskRight.size];
243  size = maskRight.size;
244  }
245 
246  for (int i = 0; i < size; i++)
247  {
248  mask[i] = maskRight.mask[i];
249  }
250  }
251 
252  return *this;
253 }

256 // Если хоть один из массивов полностью является подмножеством второго, то true. 1001 && 0001 = true; 1001 && 0011 = false
257 inline bool DYNAMIC_MASK::operator && (const DYNAMIC_MASK &maskRight)
258 {
259  //cout << "============================= DYNAMIC_MASK::operator && " << endl;
260  bool isEqual1 = true, // флаг mask1 = mask1 && mask2
261  isEqual2 = true; // флаг mask2 = mask1 && mask2
262  short size = this->size > maskRight.size ? maskRight.size : this->size;
263  for (int i = 0; i < size; i++)
264  {
265  //cout << "i=" << i << " this->mask[i]" << this->mask[i] << " maskRight.mask[i]=" << maskRight.mask[i] << endl;
266  if ( (this->mask[i] && maskRight.mask[i]) != this->mask[i]) {isEqual1 = false;}
267  if ( (this->mask[i] && maskRight.mask[i]) != maskRight.mask[i]) {isEqual2 = false;}
268  //cout << " i=" << i << " this->mask[i]" << this->mask[i] << " maskRight.mask[i]=" << maskRight.mask[i] << endl;
269  // если оба массива отличаются от результирующего, то нет смысла продолжать
270  if ( !isEqual1 && !isEqual2 )
271  {return false;}
272  }
273  return true;
274 }
