121 // Печать текста, теперь чистим массив после каждой печати
122 inline PRINTER_ERRORS printer_print(const char *str,
123  unsigned int stringSize )
124 {
125  EncodeConvertor ec;
126  string tmp; tmp.assign(str, stringSize);
127  int errP = 0;
128  tmp = ec.CP866toUTF8(tmp, &errP);
129  logINFO("printer_print() UTF8 str = \n'%s'\n", tmp.c_str());
130 
131  uint32_t sleep_time = 1;
132  for ( uint32_t i=0; i<stringSize-1; ++i )
133  {
134  if ( str[i] == '\n' )
135  ++sleep_time;
136  }
137 
138  PRINTER_ERRORS rc = ThermalPrinter_API_Print( tmp.c_str(), tmp.length() );
139 
140  // Необходимо заснуть и подождать окончание печати
141  usleep( sleep_time * stringSize * 30 );
142 
143  return rc;
144 }
