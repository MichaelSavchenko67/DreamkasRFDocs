270 bool ScaleProtocolMassak::crcPacketCreate()
271 {
272  const uint16_t MAX = 0x8000;
273  const uint16_t POLY = 0x1021;
274 
275  uint16_t base = 0;
276  uint16_t temp = 0;
277  uint16_t crcPacket = 0;
278 
279  uint16_t offset = packetFormat.headerSize + packetFormat.dateLenSize;
280  uint16_t crcPacketLen = packetFormat.cmdSize + dataSize;
281 
282  for (uint16_t i = 0; i < crcPacketLen; ++i)
283  {
284  base = 0;
285  temp = (crcPacket >> CHAR_BIT) << CHAR_BIT;
286 
287  for (uint8_t bits = 0; bits < CHAR_BIT; ++bits)
288  {
289  base = ((temp ^ base) & MAX) ? ((base << 1) ^ POLY) : (base << 1);
290  temp <<= 1;
291  }
292 
293  crcPacket = base ^ (crcPacket << CHAR_BIT) ^ (packet[offset + i] & 0xFF);
294  }
295 
296  bytes2 crcData;
297  crcData.data = crcPacket;
298  copy(crcData.bytes, crcData.bytes + sizeof(crcData.bytes), crc.get());
299 
300  return true;
301 }
