338 int FISGO_TEST_API::socket_create()
339 {
340  logDBG( "REMOTE SOCKET CREATE ..." );
341 
342  int listener = -1;
343  int enable = 1;
344 
345  // Создаем сокет
346  // AF_INET - интернет домен
347  // SOCK_STREAM - передача потоков данных с предварительной установкой соединения
348  // PROTOCOL по умолчанию
349  int PROTOCOL = 0;
350  listener = socket( AF_INET, SOCK_STREAM, PROTOCOL );
351 
352  if(listener < 0)
353  {
354  // Сокет не создан
355  logERR( "REMOTE SOCKET CREATE ERROR!!!" );
356  perror("socket");
357  return SOCK_CREATE_ERROR;
358  }
359 
360  // Использование стандартных адресов
361  if (setsockopt( listener, SOL_SOCKET, SO_REUSEPORT, &enable, sizeof(int)) < 0 )
362  {
363  // Реинит сокета невозможен
364  logERR("REMOTE SO_REUSEPORT FAILED!!!");
365  return SOCK_CREATE_ERROR;
366  }
367  // Неблокирующий сокет
368  if (fcntl( listener, F_SETFL, O_NONBLOCK ) == -1)
369  {
370  logERR( "REMOTE SOCKET NON BLOCK SET ERROR!!!" );
371  return SOCK_CREATE_ERROR;
372  }
373 
374  // Сокет создан. Возвратдескриптора
375  logDBG( "REMOTE SOCKET CREATE SUCCESS\n" );
376  return listener;
377 }

453 int FISGO_TEST_API::sock_accept( int listener )
454 {
455  int sock;
456  // Тест сокета
457  if (listener < 0)
458  {
459  logERR( "INVALID REMOTE SOCKET!!!" );
460  return ACCEPT_ERROR;
461  }
462 
463  set<int> clients;
464  clients.clear();
465 
466  while(true)
467  {
468  // Заполняем множество сокетов
469  fd_set readset;
470  FD_ZERO( &readset );
471  FD_SET( listener, &readset );
472 
473  for (set<int>::iterator it = clients.begin(); it != clients.end(); it++)
474  {
475  FD_SET( *it, &readset );
476  }
477  // Установка таймаута - 180 с
478  timeval timeout;
479  timeout.tv_sec = 180;
480  timeout.tv_usec = 0;
481  // Ждём события в одном из сокетов
482  int mx = max( listener,
483  *max_element( clients.begin(),
484  clients.end() ) );
485  logDBG( "SELECT ..." );
486  if (select( mx + 1, &readset, NULL, NULL, &timeout ) <= 0)
487  {
488  logERR( "REMOTE SELECT TIMEOUT!!!" );
489  perror("select");
490  return ACCEPT_TIMEOUT;
491  }
492  // Определяем тип события и выполняем соответствующие действия
493  if (FD_ISSET( listener, &readset ))
494  {
495  // Поступил новый запрос на соединение
496  // Используем accept
497  // Адрес клиента
498  sockaddr_in client_addr;
499  socklen_t client_addrlen = 0;
500  logDBG( "ACCEPT ..." );
501  sock = accept( listener,
502  (struct sockaddr *)&client_addr,
503  &client_addrlen );
504  if (sock < 0)
505  {
506  logERR( "REMOTE ACCEPT ERROR!!!" );
507  perror("accept");
508  return ACCEPT_ERROR;
509  }
510  if (client_addrlen != 0)
511  {
512  // Извлекаем адрес клиента
513  logDBG( "CLIENT ADDRESS: %s : %d",
514  inet_ntoa( client_addr.sin_addr ),
515  ntohs( client_addr.sin_port ) );
516  }
517  // Установка неблокирующего сокета
518  fcntl( sock, F_SETFL, O_NONBLOCK );
519  // Запись в таблицу клиентов
520  clients.insert( sock );
521  break;
522  }
523  }
524 
525  logDBG( "REMOTE CLIENT ACCEPT SUCCESS" );
526  return sock;
527 }

592  return LISTEN_ERROR;
593  }
594 
595  logDBG( "SERVER INITIALISATION COMPLETE" );

597
