158 void g_log_buf_printf( const int pri, const char *const file,
159  size_t line, const char *const fn,
160  const char *format, unsigned char *buf, unsigned int len )
161 {
162 
163  unsigned int count = len / 20 + 1;
164  unsigned int SIZE = 0;
165  SIZE = count + (FILE_SIZE + 1 + LINE_SIZE + 1 + FN_SIZE) * count + 20 * 5;
166  char buf_new[SIZE];
167  char buf_tmp[10];
168  memset( buf_new, 0 , sizeof(buf_new) );
169 
170  // Сформировать новую строку формата
171  sprintf( buf_new, "%.*s:%u:%.*s():\n", FILE_SIZE, file, line, FN_SIZE, fn );
172  for (unsigned i = 0; i < len; i++)
173  {
174  sprintf( buf_tmp, format, buf[i] );
175  strcat( buf_new, buf_tmp );
176  if ((i + 1) % 20 == 0)
177  {
178  strcat( buf_new, "\n" );
179  syslog( pri, "%s", buf_new );
180  memset( buf_new, 0 , sizeof(buf_new) );
181  sprintf( buf_new, "%.*s:%u:%.*s():\n", FILE_SIZE, file, line, FN_SIZE, fn );
182  }
183  }
184  strcat( buf_new, "\n" );
185  syslog( pri, "%s", buf_new );
186 }
