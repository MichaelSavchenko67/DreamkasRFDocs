234 void Positions::calc_total_sum_receipt( unsigned int cur_pos )
235 {
236  // Расчёт чека
237  logDBG( "CALC TOTAL SUM RECEIPT ..." );
238 
239  if(!_taxes)
240  {
241  logERR("%s Null ptr!", __PRETTY_FUNCTION__);
242  return;
243  }
244 
245  uint64_t total_sum = 0;
246  uint64_t sum_pos = 0;
247  uint64_t sum_pos_prepayment = 0;
248  uint64_t total_taxes[Taxes::MAX_TAXES+1];
249  uint64_t excess_value = 0;
250  total_sum_display = 0;
251 
252  if(!_taxes)
253  {
254  logERR("%s Null ptr!", __PRETTY_FUNCTION__);
255  return;
256  }
257 
258  memset(total_taxes, 0, sizeof(total_taxes));
259 
260  for ( Position &pos : positions_list)
261  {
262  if (((pos.goods.goods_type & GoodsData::WEIGHT) == 0) && (pos.discount == 0))
263  {
264  sum_pos = pos.get_pos_price() * pos.get_quantity();
265  pos.set_total_sum_position( sum_pos );
266  }
267  else
268  {
269  sum_pos = pos.get_total_sum_position();
270  }
271 
272  sum_pos_prepayment = pos.prepayment;
273  pos.set_tax_sum( _taxes->calc_tax_sum(E_TAXES( pos.goods.tax_number), sum_pos) );
274  total_sum += sum_pos;
275 
276  if( !pos.discountName.empty() )
277  {
278  if( pos.discountType == 1 )
279  {
280  sum_pos -= pos.discountValue;
281  }
282  else
283  {
284  sum_pos -= Convert::doubleToUint64ArithmeticRound( (sum_pos * pos.discountValue/100.) / 100 );
285  }
286  }
287  sum_pos += pos.marginSumPubl;
288 
289  if((pos.goods.goods_type & GoodsData::WEIGHT) == 0)
290  {
291  pos.set_total_sum_position( sum_pos );
292  }
293 
294  // Специальные формы оплаты
295  // Предоплата
296  if (pos.clc_prepayment)
297  {
298  logDBG( "clc_prepayment" );
299  total_sum_display += sum_pos_prepayment;
300  }
301  else
302  {
303  total_sum_display += sum_pos;
304  }
305 
306  if (pos.clc_credit)
307  {
308  logDBG( "clc_credit" );
309  total_sum_display -= pos.credit;
310  }
311 
312  if (pos.clc_full_credit)
313  {
314  logDBG( "clc_full_credit" );
315  total_sum_display -= sum_pos;
316  }
317  total_sum += sum_pos;
318  total_taxes[pos.goods.tax_number] += pos.get_tax_sum();
319 
320  if ( pos.get_pos_num() == int(cur_pos) )
321  {
322  pos.sum_pos_display = pos.get_pos_price();
323  // Специальные формы оплаты
324  // Предоплата
325  if (pos.clc_prepayment)
326  {
327  pos.sum_pos_display = pos.prepayment;
328  }
329  // Кредит
330  if (pos.clc_credit)
331  {
332  pos.sum_pos_display -= pos.credit;
333  }
334  // Кредит 100%
335  if (pos.clc_full_credit)
336  {
337  pos.sum_pos_display = 0;
338  }
339  }
340  }
341 
342  // Скидка на чек
343  if( !discountName.empty() )
344  {
345  const uint8_t ABSOLUTE = 1;
346  if( discountType == ABSOLUTE )
347  {
348  total_sum_display -= discountValue;
349  }
350  else
351  {
352  total_sum_display -= Convert::doubleToUint64ArithmeticRound( (total_sum_display * discountValue/100.) / 100 );
353  }
354  }
355  // Надбавка на чек
356  total_sum_display += marginSumPubl;
357 
358  // Зачёт ранее внесённой суммы
359  if (clc_advance)
360  {
361  logDBG( "clc_advance" );
362  if(advance <= total_sum_display)
363  {
364  excess = false;
365  credit_pay = false;
366  total_sum_display -= advance;
367  }
368  else
369  {
370  // Превышение над стоимостью
371  if (!clc_credit_pay)
372  {
373  logDBG( "excess" );
374  excess = true;
375  }
376  credit_pay = false;
377  excess_of_cost.goods.goods_name = excess_name;
378  excess_value = advance - total_sum_display;
379  total_taxes[nds_18] += _taxes->calc_tax_sum( nds_18, excess_value );
380  excess_of_cost.set_pos_num( positions_list.size() + 1 );
381  excess_of_cost.set_quantity( 1 );
382  excess_of_cost.goods.unit_price = excess_value;
383  excess_of_cost.set_pos_price( excess_value );
384  excess_of_cost.set_total_sum_position( excess_value );
385  excess_of_cost.goods.tax_number = nds_18;
386  excess_of_cost.set_tax_sum(_taxes->calc_tax_sum( nds_18, excess_value ));
387  excess_of_cost.goods.goods_type = GOODS;
388  if (!clc_credit_pay)
389  {
390  total_sum_display = 0;
391  total_sum += (advance - total_sum);
392  }
393  }
394  }
395 
396  // Оплата кредита
397  if (clc_credit_pay)
398  {
399  // Сумма оплаты кредита
400  logDBG( "clc_credit_pay" );
401  excess = false;
402  credit_payment.goods.goods_name = credit_payment_name;
403  credit_payment.set_pos_num( positions_list.size() + 1 );
404  credit_payment.set_quantity( 1 );
405  credit_payment.set_pos_price( credit_pay_sum );
406  credit_payment.set_total_sum_position( credit_pay_sum );
407  credit_payment.goods.tax_number = nds_18;
408  credit_payment.set_tax_sum(_taxes->calc_tax_sum( nds_18, credit_pay_sum ));
409  total_sum_display = credit_pay_sum;
410  }
411  set_total_tax_sum(total_taxes);
412  set_total_sum_receipt(total_sum);
413 }
