55 // Рекурсивный поиск итема в поддереве по ID
56 MenuItem* MenuItem::FindItmById(Lcd::MENU_ID searchingID)
57  {
58  // Если этот элемент и есть искомый, то возвращаем указатель на него
59  if (this->GetID() == searchingID)
60  {
61  return this;
62  }
63  // Если это не тот элемент, и это конец, то возвращаем NULL(тупик)
64  if(this->GetChildVect().empty())
65  {
66  return NULL;
67  }
68  // Если у этого элемента есть потомки, то рекурсивно ищем в них
69  for(unsigned int i=0; i<GetChildVect().size(); i++)
70  {
71  MenuItem* iterItm = GetChildVect().at(i)->FindItmById(searchingID);
72  // Если Элемент нашелся, то возращаем указатель
73  if( iterItm != NULL)
74  return iterItm;
75  }
76  // Если Элемент НЕ нашелся, то возращаем NULL
77  return NULL;
78  }
